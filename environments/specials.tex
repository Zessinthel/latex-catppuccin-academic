% Dificultad (comando)
\newcommand{\estrellita}[1]{%
  \foreach \i in {1,...,5}{%
    \ifnum\i>#1
      {\small\ding{73}}% estrella vacía (☆)
    \else
      {\small\ding{72}}% estrella rellena (★)
    \fi
    \ifnum\i<5\thinspace\fi
  }%
}

\newcommand{\dificultad}[2]{%
  \begin{tcolorbox}[
    enhanced,
    breakable,
    % interior hidden,           % comentado para ver el fondo
    colframe=CtpLavender,            % borde
    colback=CtpLavender!0,            % fondo
    arc=0.5pt,
    outer arc=0.5pt,
    boxrule=0.5pt,                  
    left=2pt,                    
    right=2pt,
    top=2pt,
    bottom=2pt,
    drop fuzzy shadow=black!5,
    fontupper=\small
  ]
  \textbf{Nivel:}\quad
  Matemática:~\estrellita{#1}\qquad
  Física:~\estrellita{#2}
  \end{tcolorbox}
}
